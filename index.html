Q.Find the Smaller Angle

    var Minimal_Angle = (h, m) => 
{
  if(h==12 && m==0)return 0
  let ma=m*6
  let ha=h*30 + m*0.5
  if(ha>ma)return(ha-ma)
  else return(ma-ha)
};

Q. perfect no.check

    var Perfect_Check = (N) => 
{
  let total=0
  if(N==1)
  return "YES"
  else{
  for(let i = 1; i < N; i++)
  {
    if(N%i == 0)
    total += i;
  }
  if(total==N)return "YES"
  else return "NO"}
};

    var Substring_Check = (S1, S2) => 
{

    if(s[i]==S2) tr=true
